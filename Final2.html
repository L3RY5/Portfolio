def haversine_np(lon1, lat1, lon2, lat2):
    """
    Calculate the great circle distance between two points
    on the earth (specified in decimal degrees)

    All args must be of equal length.    

    """
    lon1, lat1, lon2, lat2 = map(np.radians, [lon1, lat1, lon2, lat2])

    dlon = lon2 - lon1
    dlat = lat2 - lat1

    a = np.sin(dlat/2.0)**2 + np.cos(lat1) * np.cos(lat2) * np.sin(dlon/2.0)**2

    c = 2 * np.arcsin(np.sqrt(a))
    km = 6367 * c
    return km

def create_sorted_table_with_distance(data, person_lat, person_lon):
    d = pd.DataFrame()
    
    # Maak een nieuwe tabel aan met distance
    for index, row in data.iterrows():
        distance = haversine_np(person_lon, person_lat, row['long'], row['lat'])
        temp = pd.DataFrame({
            'Adres': row['Adres'], 
            'Plaatsen': row['Plaatsen'], 
            'lat': row['lat'], 
            'long': row['long'], 
            'Beschikbaar': row['Beschikbaar'], 
            'Distance': distance
        }, index=[0])
        d = pd.concat([d, temp])
    
    # Sorteren op distance ASC
    d = d.sort_values('Distance', ascending=True)
    d = d.reset_index(drop=True)
    
    return d

def create_map(radius, center_lat, center_lon, zoomstart, person_lat, person_lon, street_name, data, realtime):
    
    # Radius in km
    _radius = radius

    kaart = folium.Map(
        location=[center_lat, center_lon], 
        zoom_start = zoomstart
    )

    # Make marker for person
    _person_lat = person_lat
    _person_long = person_lon
    folium.Marker(
        [person_lat, person_lon],
        popup = street_name,
        icon = folium.Icon(color='blue')
    ).add_to(kaart)
    
    d = create_sorted_table_with_distance(data, person_lat, person_lon)
    
    firstAvailable = False
    for index, row in d.iterrows():
        if index < 65:
            if row['Distance'] < radius:
                distance = row['Distance']
                if round(distance, 1) < 1:
                    distance = distance * 1000
                    distance = str(int(distance)) + ' meter'
                else:
                    distance = str(round(distance, 2)) + ' km'
                                   
                if realtime:
                    if row['Beschikbaar'] == 1:
                        if firstAvailable == False:
                            folium.RegularPolygonMarker(
                                [row['lat'], row['long']],
                                popup = row['Adres'] + ' -- ' + distance + ' is Beschikbaar',
                                fill_color='#b511af',
                                number_of_sides=8,
                                radius=20
                                ).add_to(kaart)
                            firstAvailable = True
                        else:
                            folium.Marker(
                                [row['lat'], row['long']],
                                popup = row['Adres'] + ' -- ' + distance + ' is Beschikbaar',
                                icon = folium.Icon(icon='cloud', color="green")
                            ).add_to(kaart)
                    else:
                        folium.Marker(
                            [row['lat'], row['long']],
                            popup = row['Adres'] + ' -- ' + distance + ' is niet Beschikbaar',
                            icon = folium.Icon(icon='cloud', color="red")
                        ).add_to(kaart)
                else:
                    folium.Marker(
                        [row['lat'], row['long']],
                        popup = row['Adres'] + ' -- ' + distance,
                        icon = folium.Icon(icon='cloud')
                    ).add_to(kaart)
    return kaart

def calculate_availability(data, radius):
    total_rows = 0
    beschikbaar_count = 0
    
    for index, row in data.iterrows():
            if index < 65:
                if row['Distance'] < radius:
                    if row['Beschikbaar'] == 1:
                        beschikbaar_count = beschikbaar_count + 1
                        total_rows = total_rows + 1
                    else:
                        total_rows = total_rows + 1
    
    if total_rows == 0:
        return 0;
    else:
        percentage = (beschikbaar_count/total_rows) * 100
        dictionary = {
            'rows': total_rows,
            'beschikbaar': beschikbaar_count,
            'percentage': percentage
        }
        return dictionary
    
def print_percentages(dictionary):
    if dictionary == 0:
        print('\tEr zijn geen parkeerplekken voor gehandicapten binnen de gekozen straal.')
    else:
        print('\tTotale parkeerplekken binnen straal: ' + str(dictionary["rows"]) + 
          '\n\tAantal beschikbare parkeerplekken binnen straal: ' + str(dictionary["beschikbaar"]) + 
          '\n\tPercentage beschikbare parkeerplekken t.o.v. alle parkeerplekken binnen de straal: ' + str(round(dictionary["percentage"], 2)) + '%')
